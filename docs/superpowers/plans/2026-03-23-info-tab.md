# Info Tab con Auto-Update — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Aggiungere un quinto tab "Info" al companion app Flutter macOS con visualizzazione versione app, dati utente autenticato, URL server, e meccanismo di auto-update via GitHub Releases.

**Architecture:** Tre layer separati — `version_utils.dart` (logica semver pura, testabile), `InfoNotifier` + `UpdateService` (business logic con dependency injection per http e versione), `InfoScreen` (UI). Il tab è wired in `app_router.dart` e `home_shell.dart`. L'auto-update scarica il DMG da GitHub Releases, genera uno script bash in `/tmp` con path passati come env-var, e termina l'app lasciando lo script girare in background.

**Tech Stack:** Flutter 3.x, Riverpod 2.x (`StateNotifierProvider`), go_router (`StatefulShellRoute`), `package_info_plus`, `dart:io` (`Process`), `http`, `mocktail` per i test

**Spec:** `docs/superpowers/specs/2026-03-23-info-tab-design.md`

```dart
/// Restituisce true se [latest] è una versione maggiore di [installed].
/// Ignora i tag pre-release (tag che contengono '-', es. "1.1.0-beta").
/// Confronto numerico per segmento — non lessicografico.
bool isNewerVersion(String installed, String latest) {
  if (latest.contains('-')) return false;

  final a = installed.split('.').map(int.parse).toList();
  final b = latest.split('.').map(int.parse).toList();

  for (var i = 0; i < b.length; i++) {
    final ai = i < a.length ? a[i] : 0;
    if (b[i] > ai) return true;
    if (b[i] < ai) return false;
  }
  return false;
}
```

```dart
import 'dart:convert';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:http/http.dart' as http;
import 'package:package_info_plus/package_info_plus.dart';
import '../../core/api/api_client.dart';
import '../../core/auth/auth_provider.dart';
import '../../core/models/user_profile.dart';
import '../../core/utils/version_utils.dart';

enum UpdateStatus { idle, checking, upToDate, updateAvailable, downloading, error }

class InfoState {
  final UserProfile? profilo;
  final UpdateStatus updateStatus;
  final String? versione;
  final String? nuovaVersione;
  final String? dmgUrl;
  final String? serverUrl;
  final double? downloadProgress;
  final String? errore;

  const InfoState({
    this.profilo,
    this.updateStatus = UpdateStatus.idle,
    this.versione,
    this.nuovaVersione,
    this.dmgUrl,
    this.serverUrl,
    this.downloadProgress,
    this.errore,
  });

  InfoState copyWith({
    UserProfile? profilo,
    UpdateStatus? updateStatus,
    String? versione,
    String? nuovaVersione,
    String? dmgUrl,
    String? serverUrl,
    double? downloadProgress,
    String? errore,
    bool clearErrore = false,
    bool clearProgress = false,
  }) =>
      InfoState(
        profilo: profilo ?? this.profilo,
        updateStatus: updateStatus ?? this.updateStatus,
        versione: versione ?? this.versione,
        nuovaVersione: nuovaVersione ?? this.nuovaVersione,
        dmgUrl: dmgUrl ?? this.dmgUrl,
        serverUrl: serverUrl ?? this.serverUrl,
        downloadProgress: clearProgress ? null : (downloadProgress ?? this.downloadProgress),
        errore: clearErrore ? null : (errore ?? this.errore),
      );
}

const _githubApiUrl =
    'https://api.github.com/repos/wtechdev/maelstrom_companion/releases/latest';

class InfoNotifier extends StateNotifier<InfoState> {
  final ApiClient _apiClient;
  final http.Client _httpClient;
  final Future<String> Function() _getInstalledVersion;
  bool _checkFatto = false;

  InfoNotifier(
    this._apiClient, {
    http.Client? httpClient,
    Future<String> Function()? getVersion,
  })  : _httpClient = httpClient ?? http.Client(),
        _getInstalledVersion = getVersion ?? _defaultGetVersion,
        super(const InfoState());

  Future<void> init() async {
    if (_checkFatto) return;
    _checkFatto = true;
    await _caricaProfilo();
    await checkUpdate();
  }

  Future<void> _caricaProfilo() async {
    try {
      final profilo = await _apiClient.getProfilo();
      state = state.copyWith(profilo: profilo, serverUrl: _apiClient.baseUrl);
    } catch (_) {
      state = state.copyWith(serverUrl: _apiClient.baseUrl);
    }
  }

  Future<void> checkUpdate() async {
    state = state.copyWith(updateStatus: UpdateStatus.checking, clearErrore: true);
    try {
      final installedVersion = await _getInstalledVersion();
      state = state.copyWith(versione: installedVersion);

      final response = await _httpClient
          .get(Uri.parse(_githubApiUrl), headers: {'Accept': 'application/vnd.github+json'})
          .timeout(const Duration(seconds: 10));

      if (response.statusCode == 403 || response.statusCode == 429) {
        state = state.copyWith(
          updateStatus: UpdateStatus.error,
          errore: 'Limite richieste GitHub raggiunto. Riprova tra un\'ora.',
        );
        return;
      }
      if (response.statusCode != 200) {
        state = state.copyWith(
          updateStatus: UpdateStatus.error,
          errore: 'Errore GitHub: ${response.statusCode}',
        );
        return;
      }

      final json = jsonDecode(response.body) as Map<String, dynamic>;
      final tagName = json['tag_name'] as String;
      final latestVersion = tagName.replaceFirst('v', '');

      final assets = json['assets'] as List<dynamic>? ?? [];
      final dmgAsset = assets.cast<Map<String, dynamic>>().firstWhere(
            (a) => (a['name'] as String).endsWith('.dmg'),
            orElse: () => {},
          );
      final dmgUrl = dmgAsset['browser_download_url'] as String?;

      if (isNewerVersion(installedVersion, latestVersion)) {
        state = state.copyWith(
          updateStatus: UpdateStatus.updateAvailable,
          nuovaVersione: latestVersion,
          dmgUrl: dmgUrl,
        );
      } else {
        state = state.copyWith(updateStatus: UpdateStatus.upToDate);
      }
    } on Exception catch (e) {
      state = state.copyWith(
        updateStatus: UpdateStatus.error,
        errore: 'Impossibile controllare aggiornamenti: $e',
      );
    }
  }

  void aggiornaProgress(double progress) {
    state = state.copyWith(updateStatus: UpdateStatus.downloading, downloadProgress: progress);
  }

  void impostaErrore(String messaggio) {
    state = state.copyWith(updateStatus: UpdateStatus.error, errore: messaggio);
  }
}

  final info = await PackageInfo.fromPlatform();
  return info.version;
}
```

```dart
import 'dart:io';
import 'package:http/http.dart' as http;

/// Gestisce il download del DMG di aggiornamento e la generazione/esecuzione
/// dello script bash di sostituzione.
class UpdateService {
  final http.Client _httpClient;

  UpdateService({http.Client? httpClient})
      : _httpClient = httpClient ?? http.Client();

  /// Path di destinazione del DMG scaricato.
  static String dmgPath(String version) =>
      '/tmp/MaelstromCompanion-$version.dmg';

  /// Path del bundle app in esecuzione.
  /// Deriva da Platform.resolvedExecutable risalendo di 3 livelli:
  ///   .../Maelstrom Companion.app/Contents/MacOS/binary → .../Maelstrom Companion.app
  static String get appInstallPath {
    final exe = Platform.resolvedExecutable;
    final parts = exe.split('/');
    if (parts.length > 3) {
      return parts.sublist(0, parts.length - 3).join('/');
    }
    return '/Applications/Maelstrom Companion.app';
  }

  /// Genera il contenuto dello script bash. I path vengono letti da variabili
  /// di ambiente (non interpolati nel codice) per evitare shell injection.
  static String generaScript(String dmgPath, String appPath) => '''
#!/bin/bash
# Variabili passate come env-var dal processo chiamante
DMG_PATH="\${MAELSTROM_DMG_PATH}"
APP_DEST="\${MAELSTROM_APP_DEST}"

sleep 2

# Monta il DMG
MOUNT=\$(hdiutil attach "\$DMG_PATH" -nobrowse -quiet | tail -1 | awk '{print \$NF}')
if [ -z "\$MOUNT" ]; then exit 1; fi

# Sostituisce il bundle (ditto preserva xattr e fork risorse)
rm -rf "\$APP_DEST"
ditto "\$MOUNT/Maelstrom Companion.app" "\$APP_DEST"

# Smonta e rilancia
hdiutil detach "\$MOUNT" -quiet
open "\$APP_DEST"

# Pulizia
rm -f "\$DMG_PATH"
rm -f "\$0"
''';

  /// Scarica il DMG da [url] in [destinazione], invocando [onProgress]
  /// con un valore 0.0–1.0 durante il download.
  Future<void> scaricaDmg({
    required String url,
    required String destinazione,
    required void Function(double) onProgress,
  }) async {
    final request = http.Request('GET', Uri.parse(url));
    final response = await _httpClient.send(request);

    final contentLength = response.contentLength ?? 0;
    var received = 0;

    final file = File(destinazione);
    final sink = file.openWrite();
    try {
      await for (final chunk in response.stream) {
        sink.add(chunk);
        received += chunk.length;
        if (contentLength > 0) {
          onProgress(received / contentLength);
        }
      }
    } finally {
      await sink.close();
    }
  }

  /// Scrive lo script di aggiornamento in /tmp, lo rende eseguibile,
  /// lo avvia con le variabili di ambiente corrette, e termina l'app.
  Future<void> avviaAggiornamento({
    required String dmgPath,
    required String appPath,
  }) async {
    const scriptPath = '/tmp/maelstrom_updater.sh';

    // Scrive lo script
    final scriptFile = File(scriptPath);
    await scriptFile.writeAsString(generaScript(dmgPath, appPath));

    // Rende eseguibile
    await Process.run('chmod', ['+x', scriptPath]);

    // Avvia lo script con i path come variabili di ambiente
    await Process.start(
      scriptPath,
      [],
      environment: {
        'MAELSTROM_DMG_PATH': dmgPath,
        'MAELSTROM_APP_DEST': appPath,
      },
      mode: ProcessStartMode.detached,
    );
  }
}
```

```dart
class _LogoTabItem extends StatelessWidget {
  final String label;
  final bool selected;
  final VoidCallback onTap;
  const _LogoTabItem({required this.label, required this.selected, required this.onTap});

  @override
  Widget build(BuildContext context) {
    final cs = Theme.of(context).colorScheme;
    final color = selected ? cs.primary : cs.onSurface.withValues(alpha: 0.5);
    return InkWell(
      onTap: onTap,
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          ColorFiltered(
            colorFilter: ColorFilter.mode(color, BlendMode.srcIn),
            child: const WtechLogo(height: 22, brandmarkOnly: true),
          ),
          const SizedBox(height: 2),
          Text(
            label,
            style: TextStyle(
              fontSize: 10,
              color: color,
              fontWeight: selected ? FontWeight.w600 : FontWeight.normal,
            ),
          ),
        ],
      ),
    );
  }
}
```
